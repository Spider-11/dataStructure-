#include < stdio.h >

int StrToInt(char str[]);
int main() {
    int value;
    char str[20];
    printf("\nEnter any string ");
    scanf("%s", str);
    value = StrToInt(str);
    if (value != -1) {
        printf("\nEqual integer value: %d", value);
    }
    return 0;
}
int StrToInt(char str[]) {
    int i = 0, sum = 0;
    while (str[i] != '\0') {
        if (str[i] < '0' || str[i] > '9') {
            printf("\nUnable to convert");
            return -1;
        } else {
            sum = sum * 10 + (str[i] - '0');
            i++;
        }
    }
    return sum;
}
