int splitAndReturnIndex(char str[]) {
    int i, index = 0;
    for (i = 0; str[i] == ' '; i++)
        ;
    while (str[i] != ' ' && str[i] != '\0') {
        i++;
    }
    for (; str[i] == ' '; i++)
        ;
    while (str[i] != ' ' && str[i] != '\0') {
        i++;
        index++;
    }
    return index;
}
int main() {
    char str[100];
    printf("Enter the string: ");
    gets(str);
    int index = splitAndReturnIndex(str);
    if (index > 0) {
        printf("Index of the second word: %d\n", index);
    } else {
        printf("No second word found.\n");
    }
    return 0;
}
