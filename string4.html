#include <stdio.h>
int main() {
    char input[100];
    int i = 0;
    int isFirstChar = 1; 
    printf("Enter a string: ");
    while (1) {
        char c = getchar();
        if (c == '\n') {
            break;
        }
        if (isFirstChar && (c >= 'a' && c <= 'z')) {
            c -= 32; 
            isFirstChar = 0;
        } else if (!isFirstChar && (c >= 'A' && c <= 'Z')) {
            c += 32;
        }
        input[i++] = c;
       
        if (c == ' ' || c == '\t') {
            isFirstChar = 1;
        }
    }
    printf("Title case: %s\n", input);
    return 0;
}
