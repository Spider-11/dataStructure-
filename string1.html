#include <conio.h>

    char str[10], i;

    printf("Enter the string\n");
    gets(str);

    for (i = 0; str[i]; i++) {
        if (str[i] >= 'a' && str[i] <= 'z') {
            str[i] = str[i] - 32;
        }
    }

    printf("Uppercase string is %s", str);
}
