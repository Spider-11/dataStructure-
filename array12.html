#include <stdio.h>


void duplicate(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        if (arr[i] != 0) {
 for (int j = i + 1; j < size; j++) {
 if (arr[j] == arr[i]) {
 arr[j] = 0;
 }
 }
 }
    }
}

int main() {
    int arr[10];
    printf("Enter the elements of the array:\n");
    for (int i = 0; i < 10; i++) {
        scanf("%d", &arr[i]);
    }
    duplicate(arr, 10);
    printf("Processed Array:\n");
    for (int i = 0; i < 10; i++) {
        printf("%d ", arr[i]);
    }
}
