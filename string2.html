#include <stdio.h>
#include <conio.h>

int main() {
    char input_string[100];
    printf("Enter a string: ");
    scanf("%s", input_string);
    for (int i = 0; input_string[i] != '\0'; i++) {
        if (input_string[i] >= 'A' && input_string[i] <= 'Z') {
            input_string[i] += 32;
        }
    }
    printf("String in lowercase: %s\n", input_string);
    return 0;
}
