#include <stdio.h>

int main() {
    int arr[8], i, max, min;
    printf("Enter the array:\n");
    for (i = 0; i < 8; i++) {
        scanf("%d", &arr[i]);
    }
    max = arr[0];
    min = arr[0];
    for (i = 0; i < 8; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
        if (arr[i] < min) {
            min = arr[i];
        }
    }
    printf("Maximum value: %d\n", max);
    printf("Minimum value: %d", min);
    return 0;
}
