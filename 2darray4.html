 #include
int main()
{
	int first[3][3],second[3][3];
	int result[3][3];
	int i,j;
	printf("Enter the element of the first matrix:\n ");
	for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)
		{
			scanf("%d",&first[i][j]);
		}
	}
	printf("Enter the element of the second matrix:\n ");
	for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)
		{
			scanf("%d",&second[i][j]);
		}
	}
		for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)
		{
			result[i][j]=first[i][j]-second[i][j];
		}
	}
	
	printf("subtraction of two matrix is:\n");
		for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)
		{
			printf("%d\t",result[i][j]);
		}
		printf("\n");
	}
	
}
