int palin(char str2[], char str[]) {
    int i;
    for (i = 0; str2[i]; i++) {
        if (str[i] != str2[i]) {
            return -1;
        }
    }
    return 1; 
}

    char str[10], str2[10], i;

    fgets(str, sizeof(str), stdin);

    // Calculate the length of the input string
    while (str[len] != '\0') {
        len++;
    }

    // Reverse the string
    for (i = len - 1; i >= 0; i--) {
        str2[j] = str[i];
        j++;
    }
    str2[j] = '\0';

    // Check if it's a palindrome
    k = palin(str2, str);

    if (k == 1) {
        printf("String is palindrome\n");

        printf("String is not palindrome\n");

}
