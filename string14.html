#include < stdio.h >

int main() {
    char str[100];
    int i;
    printf("Enter a string: ");
    scanf("%s", str);
    printf("ASCII differences:\n");
    for (i = 0; str[i + 1] != '\0'; i++) {
        int diff = str[i + 1] - str[i];
        printf("%c - %c = %d\n", str[i + 1], str[i], diff);
    }
    return 0;
}
