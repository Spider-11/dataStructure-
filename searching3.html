#include <stdio.h> 
#include <conio.h>

int linear(int ar[], int i, int item) {
    if (i >= 10) {
        return -1; // Base case: item not found in the array
    }

    if (ar[i] == item) {
        return i; // Item found at index i
    } else {
        return linear(ar, i + 1, item); // Recursive call for the next index
    }
}

int main() {
    int ar[10], item, i, search;

    printf("Enter the arrays\n");
    for (i = 0; i < 10; i++) {
        printf("Enter the array at location %d\t", i + 1);
        scanf("%d", &ar[i]);
    }

    printf("Enter the item you want to search\n");
    scanf("%d", &item);

    search = linear(ar, 0, item);

    if (search != -1) {
        printf("Item is found at location %d\t", search + 1);
    } else {
        printf("Item is not in the list\n");
    }

    return 0;
}
