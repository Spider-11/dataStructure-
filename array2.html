   #include <stdio.h>
int main()    
{ 
    int arr[5];        
    printf("Original array: \n");    
    for (int i = 0; i < 5; i++) {     
        scanf("%d",&arr[i]);    
    }      
    printf("Array in reverse order: \n");    
    for (int i = 5-1; i >= 0; i--) {     
        printf("%d ", arr[i]);     
    }     
}
