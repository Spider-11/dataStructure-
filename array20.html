#include <stdio.h>



void main()
{
    int n=10,arr[10],i,left[10],right[10],min,water;
    for ( i = 0; i < 10; i++)
    {
        printf("Enter value at %d position\n",i+1);
        scanf("%d",&arr[i]);
    }
   left[0]=arr[0];
   for ( i = 1; i < 10; i++)
   {
     if(left[i-1] < arr[i])
        left[i]=arr[i];
     
     else
     left[i]=left[i-1];
   }
   right[n-1]=arr[n-1];
   for(i=n-1;i > =0;i--)
   {
    if(right[i+1] < arr[i])
    right[i]=arr[i];
    else
    right[i]=right[i+1];
   }
  int ans=0;
  for ( i = 0; i < n; i++)
  {
    if(left[i] < right[i])
    min=left[i];
    else
    min=right[i];
    water=min-arr[i];
    ans+=water;
  }
  printf("tapped water %d",ans);
  
}
