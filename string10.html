#include < stdio.h >
int main() {
    char str[100]; 
    int position; 
	char character;
    printf("Enter a string: ");
    scanf("%s", str);
    printf("Enter the position : ");
    scanf("%d", &position);
    int length = 0;
    while (str[length] != '\0') {
        length++;
    }
    if (position >= 0 && position < length) {
         character = str[position];
        printf("Character at position %d: %c\n", position, character);
    } else {
        printf("Invalid position.\n");
    }
    return 0;
}
