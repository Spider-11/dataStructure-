float StrToFloat(char str[])
{
	int i=0;
	float sum=0.0;
	float decimal=0.1;
	 while (str[i] != '\0') {
        if ((str[i] >= '0' && str[i] <= '9') || str[i] == '.') {
            if (str[i] == '.') {
              
                decimal = 0.1;
            } else {
                if (decimal == 0.1) {
                 
                    sum = sum + (str[i] - '0') * decimal;
                    decimal /= 10;
                } else {
                   
                    sum = sum * 10 + (str[i] - '0');
                }
            }
            i++;
        } else {
            printf("\n Unable to convert");
            return -1.0;
        }
    }

    return sum;
}
