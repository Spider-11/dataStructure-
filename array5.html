#include <stdio.h>



int main() {
    int arr[10], i;
    printf("Enter the array:\n");
    for (i = 0; i < 10; i++) {
        scanf("%d", &arr[i]);
}
    printf("Even numbers in the array are:\n");
    for (i = 0; i < 10; i++) {
        if (arr[i] % 2 == 0) {
            printf("%d\n", arr[i]);
        }
    }
    return 0;
}
