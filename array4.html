#include <stdio.h>

int main() {
    int size = 5;
    int arr[size], reversedArr[size];

    printf("Enter the elements of the array:\n");
    for (int i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }

    for (int i = 0; i < size; i++) {
        reversedArr[i] = arr[size - 1 - i];
    }

    printf("\nReversed Array:\n");
    for (int i = 0; i < size; i++) {
        printf("%d ", reversedArr[i]);
    }

    return 0;
}
