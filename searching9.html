#include <stdio.h> 
#include <time.h>


int linear(int arr[], int size, int key) {
    int i, f = 0;

    for (i = 0; i < size; i++) {
        if (arr[i] == key) {
            f = 1;
            break;
        }
    }

    if (f == 1) {
        return i;
    } else {
        return -1;
    }
}

int main() {
    int key, arr[10], i, index;
    clock_t start, end;
    double cpu_time_used;

    printf("Enter the arrays\n");

    for (i = 0; i < 10; i++) {
        scanf("%d", &arr[i]);
    }

    printf("Enter the element\n");
    scanf("%d", &key);

    start = clock();
    index = linear(arr, 10, key);
    end = clock();

    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;

    if (index != -1) {
        printf("Element found at index %d\n", index + 1);
    } else {
        printf("Element is not found\n");
    }

    if (index == 0) {
        printf("Best complexity of linear search\n");
    } else {
        printf("Worst complexity of linear search\n");
    }

    printf("Time taken for search: %lf seconds\n", cpu_time_used);

    return 0;
}
