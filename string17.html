#include < stdio.h >
#include < conio.h >

int main() {
    char str[30], i;
    int count = 0;

    printf("Enter the string\n");
    gets(str);

    for (i = 0; str[i]; i++) {
        if (str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u') {
            count++;
        }
    }

    printf("Entered String is %s\n", str);
    printf("Total vowels in strings are %d", count);

    return 0;
}
