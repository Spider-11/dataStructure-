#include < stdio.h >
int main() {
    char str[100], ch; 
    printf("Enter a string: ");
    scanf("%s", str);
    printf("Enter a character to search for: ");
    scanf(" %c", &ch); 
    int position = -1;
    for (int i = 0; str[i] != '\0'; i++) {
        if (str[i] == ch) {
            position = i; 
            break; 
        }
    }
    if (position != -1) {
        printf("Character '%c' found at position %d in the string.\n", ch, position);
    } else {
        printf("Character '%c' not found in the string.\n", ch);
    }
    return 0;
}
