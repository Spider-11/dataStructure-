#include < stdio.h >


int main() {
    char str[10], temp;
    int i, j, len = 0;

    printf("Enter the string\n");
    gets(str);

    for (i = 0; str[i]; i++) {
        len++;
    }

    printf("%d", len);

    for (i = 0; i < len; i++) {
        for (j = i + 1; j < len; j++) {
            if (str[i] > str[j]) {
                temp = str[i];
                str[i] = str[j];
                str[j] = temp;
            }
        }
    }

    printf("Sorted String is %s", str);

    return 0;
}
