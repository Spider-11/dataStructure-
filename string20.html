#include < stdio.h >
int main() {
    char str[300];
    int i, len = 300;
    int space = 0, line = 0, ch = 0, word = 1;

    printf("Enter the string\n");
    gets(str);

    for (i = 0; str[i]; i++) {
        if (str[i] == ' ') {
            space++;
            word++;
        } else if (str[i] == '\n') {
            line++;
            word++;
        } else if (str[i] != '\0') {
            ch++;
        }
    }

    printf("space:%d word:%d line:%d character:%d", space, word, line, ch);

    return 0;
}
