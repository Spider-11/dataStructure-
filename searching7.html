#include <stdio.h> 
#include <conio.h>
int binary(int ar[], int item, int beg, int end) {
    int mid;

    if (beg <= end) {
        mid = (beg + end) / 2;

        if (ar[mid] == item) {
            return mid;
        }

        if (ar[mid] < item) {
            return binary(ar, item, mid + 1, end);
        } else {
            return binary(ar, item, beg, mid - 1);
        }
    }

    return -1;
}

int main() {
    int ar[10], item, i, search, beg = 0, end = 9;

    printf("Enter the arrays\n");

    for (i = 0; i < 10; i++) {
        printf("Enter the array at location %d\t", i + 1);
        scanf("%d", &ar[i]);
    }

    // Assuming the array is sorted, you might want to check and sort it if needed.

    printf("Enter the item you want to search\n");
    scanf("%d", &item);

    search = binary(ar, item, beg, end);

    if (search >= 0 && search < 10) {
        printf("Item is searched at location %d\t", search + 1);
    } else {
        printf("Item is not in the list\n");
    }

    return 0;
}
